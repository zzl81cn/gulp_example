export default class IndexModel {
    /**

    constructor(app){}
    /**
     * 获取具体数据api接口
     * @return {Promise} 返回异步数据
     * @example
     * return new Promise
     * getData()
     */
    getData(){
        return new Promise((resolve, reject) => {
            setTimeout(function () {
                resolve('异步数据')
            }, 1000)
        })
    }
}
